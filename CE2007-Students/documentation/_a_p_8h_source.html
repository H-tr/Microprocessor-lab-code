63 // UART serial from CC2650 to TM4C123/MSP432
64 // UART serial from TM4C123/MSP432 to CC2650
65 // MRDY, an output from TM4C123/MSP432 and an input to CC2650
66 // SRDY, an input to TM4C123/MSP432 and an output from CC2650
67 // Reset, an output from TM4C123/MSP432 to hardware reset to CC2650
68 // MRDY is a GPIO pin whose logic level indicates the state of the master processor.
69 // MRDY is an active low (low true) signal.
70 // An MRDY event indicates that the master processor is ready to send or receive data.
71 // SRDY is a GPIO pin whose logic level indicates the state of the slave processor.
72 // SRDY is an active low (low true) signal.
73 // An SRDY event indicates that the slave processor is ready to send or receive data.
74 // see GPIO.c file for hardware connections

213 //*************AP_AddCharacteristic**************
214 // Add a read, write, or read/write characteristic
215 // for notify properties, call AP_AddNotifyCharacteristic
216 // Inputs uuid is 0xFFF0, 0xFFF1, ...
217 // thesize is the number of bytes in the user data 1,2,4, or 8
218 // pt is a pointer to the user data, stored little endian
219 // permission is GATT Permission, 0=none,1=read,2=write, 3=Read+write
220 // properties is GATT Properties, 2=read,8=write,0x0A=read+write
221 // name is a null-terminated string, maximum length of name is 20 bytes
222 // (*ReadFunc) called before it responses with data from internal structure
223 // (*WriteFunc) called after it accepts data into internal structure
224 // Output APOK if successful,
225 // APFAIL if name is empty, more than 8 characteristics, or if SNP failure
226 int AP_AddCharacteristic(uint16_t uuid, uint16_t thesize, void *pt, uint8_t permission,
227  uint8_t properties, char name[], void(*ReadFunc)(void), void(*WriteFunc)(void));
228 
229 //*************AP_AddNotifyCharacteristic**************
230 // Add a notify characteristic
231 // for read, write, or read/write characteristic, call AP_AddCharacteristic
232 // Inputs uuid is 0xFFF0, 0xFFF1, ...
233 // thesize is the number of bytes in the user data 1,2,4, or 8
234 // pt is a pointer to the user data, stored little endian
235 // name is a null-terminated string, maximum length of name is 20 bytes
236 // (*CCCDfunc) called after it accepts , changing CCCDvalue
237 // Output APOK if successful,
238 // APFAIL if name is empty, more than 4 notify characteristics, or if SNP failure
239 int AP_AddNotifyCharacteristic(uint16_t uuid, uint16_t thesize, void *pt,
